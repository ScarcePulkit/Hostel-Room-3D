48// declaring the axis for the frames
49static void getNormal3p
50(GLfloat x1, GLfloat y1,GLfloat z1, GLfloat x2, GLfloat y2,GLfloat z2, GLfloat x3, GLfloat y3,GLfloat z3)
51{
52 GLfloat Ux, Uy, Uz, Vx, Vy, Vz, Nx, Ny, Nz;
53 //up vector
54 Ux = x2-x1;
55 Uy = y2-y1;
56 Uz = z2-z1;
57
58 Vx = x3-x1;
59 Vy = y3-y1;
60 Vz = z3-z1;
61
62 // normal vector
63 Nx = Uy*Vz - Uz*Vy;
64 Ny = Uz*Vx - Ux*Vz;
65 Nz = Ux*Vy - Uy*Vx;
66
67 glNormal3f(Nx,Ny,Nz);
68}
